Библиотека для вычисления площади поверхности различных геометрических тел.
Включает функции для расчёта куба, параллелепипеда, призмы, пирамиды, 
цилиндра, конуса и усечённого конуса.

--------------------------------------------------------------------------------

     // Прямоугольный параллелепипед
        double Cuboid(double a, double b, double c);
            a, b, c - стороны прямоугольного параллелепипеда.

    // Куб
        double Cube(double a);
            a - сторона куба.

    // Призма с правильным многоугольником в основании
        double Prism(int n, double a, double h);
            n - число сторон основания (правильный n-угольник)
            a - сторона призмы
            h - высота призмы

    // Пирамида с правильным многоугольником в основании
        double Pyramid(int n, double a, double l);
            n - число сторон основания (правильный n-угольник)
            l - образующая пирамиды
            a - сторона пирамиды
    
    // Цилиндр
        double Cylinder(double r, double h);
            r - радиус цилиндра
            h - высота цилиндра

    // Конус
        double Cone(double r, double l);
            r - радиус конуса (радиус основания)
            l - образующая конуса

    // Усечённый конус
        double TruncatedCone(double r1, double r2, double l);
            r1, r2 - верхний и нижний радиус оснований усечённого конуса
            l - образующая усечённого конуса

--------------------------------------------------------------------------------

Подключение и пример использования:

#include "geometry.h"
#include <iostream>

int main() {
    // Вычисление площади куба
    double cubeArea = geometry::Cube(5.0);
    std::cout << "Площадь куба: " << cubeArea << std::endl;
    
    // Вычисление площади цилиндра
    double cylinderArea = geometry::Cylinder(3.0, 7.0);
    std::cout << "Площадь цилиндра: " << cylinderArea << std::endl;
    
    return 0;
}
